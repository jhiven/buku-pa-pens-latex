%============================================================
% FORMAT: BUKU PROYEK AKHIR
% Aturan layout khusus Buku PA.
% Di-input SETELAH config/formatting.tex agar bisa override.
%
% Panduan: Buku PA Sarjana Terapan A4 Rev05
%============================================================

% ── Geometri: double-sided, mirror margin Rev05 ───────────────
\geometry{
  a4paper,
  twoside,
  top    = 3cm,
  bottom = 3cm,
  inner  = 4cm,   % Margin sisi jilid (kiri halaman ganjil)
  outer  = 3cm    % Margin luar
}

% ── Spasi Baris (Rev05: 1.5) ──────────────────────────────────
% Temporary latex's spacing 1.5 for cover
\onehalfspacing

% ── Indentasi Paragraf (Rev05: 1.27cm) ───────────────────────
\setlength{\parindent}{1.27cm}

% ── Override Subsubsection spacing (khusus Buku PA) ──────────
% formatting.tex pakai \titlespacing* (ada bintang) yang men-suppress
% indent paragraf pertama setelah heading. Di Buku PA ini tidak mau -
% paragraf pertama harus tetap mendapat \parindent = 1.27cm agar
% selaras dengan posisi teks subsubsection (yang juga mulai di \parindent).
% Solusi: override dengan \titlespacing TANPA bintang.
\titlespacing{\subsubsection}{\parindent}{3pt}{2pt}
